# File lib/sequel/connection_pool/single.rb, line 25
def hold(server=nil)
  begin
    unless c = @conn.first
      @conn.replace([c = make_new(:default)])
    end
    yield c
  rescue Sequel::DatabaseDisconnectError, *@error_classes => e
    disconnect if disconnect_error?(e)
    raise
  end
end

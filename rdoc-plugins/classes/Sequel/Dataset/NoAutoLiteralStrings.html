def filter_expr(expr = nil)
  case expr
  when LiteralString
    super
  when String
    raise Error, "plain string passed to a dataset filtering method"
  when Array
    if expr.first.is_a?(String) && !expr.first.is_a?(LiteralString)
      raise Error, "plain string passed to a dataset filtering method"
    end
    super
  else
    super
  end
end

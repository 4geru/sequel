def type_convertor(map, meta, type, i)
  case type
  when NUMERIC_TYPE
    if meta.getScale(i) == 0
      map[:OracleDecimal]
    else
      super
    end
  when *TIMESTAMPTZ_TYPES
    map[TIMESTAMP_TYPE]
  else
    super
  end
end

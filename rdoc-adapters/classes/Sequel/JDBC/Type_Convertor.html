def DB2Clob(r, i)
  if v = r.getClob(i)
    v = v.getSubString(1, v.length)
    v = Sequel::SQL::Blob.new(v) if ::Sequel::DB2::use_clob_as_blob
    v
  end
end

def OracleDecimal(r, i)
  if v = r.getBigDecimal(i)
    i = v.long_value
    if v == JAVA_BIG_DECIMAL_CONSTRUCTOR.call(i)
      i
    else
      BigDecimal.new(v.to_string)
    end
  end
end
